    <script>
        // *** IMPORTANT: PASTE YOUR GOOGLE APPS SCRIPT URL HERE ***
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwQ-GW915RGgJ6QHsVgkrUDFfSWHUMiwAvXA4Rh-5wAJ_hpv0TLvi9Nt82T7GHdOa3V/exec';

        const loadingState = document.getElementById('loading-state');
        const activationState = document.getElementById('activation-state');
        const foundState = document.getElementById('found-state');
        const errorState = document.getElementById('error-state');
        const errorMessage = document.getElementById('error-message');

        window.addEventListener('load', async () => {
            const urlParams = new URLSearchParams(window.location.search);
            const tagId = urlParams.get('id');

            if (!tagId) {
                showError('No Tag ID found in the URL.');
                return;
            }

            try {
                const response = await fetch(`${SCRIPT_URL}?id=${tagId}`);
                if (!response.ok) {
                    throw new Error(`Server responded with status: ${response.status}`);
                }
                const data = await response.json();

                if (data.result === 'new') {
                    showActivationForm(tagId);
                } else if (data.result === 'found') {
                    showFoundScreen(data.wa_link);
                } else {
                    throw new Error(data.message || 'Invalid response from server.');
                }
            } catch (err) {
                showError(`Failed to fetch tag status: ${err.message}`);
            }
        });

        function showActivationForm(tagId) {
            loadingState.classList.add('hidden');
            activationState.classList.remove('hidden');
            
            const form = document.getElementById('activation-form');
            form.addEventListener('submit', async (e) => {
                e.preventDefault();
                const itemName = document.getElementById('item-name').value;
                const ownerPhone = document.getElementById('owner-phone').value;

                if (!itemName || !ownerPhone) {
                    alert('Please fill out all fields.');
                    return;
                }

                try {
                    const postData = { id: tagId, item: itemName, phone: ownerPhone };
                    const response = await fetch(SCRIPT_URL, {
                        method: 'POST',
                        body: JSON.stringify(postData),
                        headers: {
                            'Content-Type': 'application/json'
                        }
                    });

                    const result = await response.json();
                    if (result.result === 'success') {
                        // Reload the page to show the "Found" screen
                        window.location.reload();
                    } else {
                        throw new Error(result.message || 'Failed to activate tag.');
                    }
                } catch (err) {
                    showError(`Activation failed: ${err.message}`);
                }
            });
        }

        function showFoundScreen(whatsappLink) {
            loadingState.classList.add('hidden');
            const linkElement = document.getElementById('whatsapp-link');
            linkElement.href = whatsappLink;
            foundState.classList.remove('hidden');
        }

        function showError(message) {
            loadingState.classList.add('hidden');
            errorMessage.textContent = message;
            errorState.classList.remove('hidden');
        }
    </script>
